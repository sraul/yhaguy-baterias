SELECT 	p.PERSONA as RAZONSOCIAL,
        P.RUC,
		P.DIRECCION,
		P.TELEFONO,
		r.DESCRIPCION AS RUBRO,
		p.IDPERSONA
FROM 	PERSONA p INNER JOIN RUBROEMPRESA r ON (p.IDRUBROEMPRESA = r.IDRUBROEMPRESA)
				  INNER JOIN CIUDAD ci on (ci.IDCIUDAD = p.IDCIUDAD)
				  INNER JOIN DEPGEOGRAFICO DPTO ON (DPTO.IDDEPARTAMENTO = CI.IDDEPARTAMENTO)
WHERE	P.ESTADO = 'A'
AND 	p.IDRUBROEMPRESA NOT IN (57,0,44,22,32,75,-1,0,1,18,21,36,38,54)
ORDER BY	3 ASC

/** RAZONSOCIAL,RUC,DIRECCION,TELEFONO,RUBRO,IDPERSONA **/
